2. Thao tác với database
"""
    sequelize db:drop // Xóa database nếu cần tạo lại
    sequelize db:drop // Tạo database nếu chưa có (không cần viết script tạo bảng trong sql)
    node db.js // Tạo bảng trong database
    sequelize db:migrate // Không cần chạy lệnh này
    sequelize db:seed:all // Tạo dữ liệu mẫu
"""

3. Xóa dữ liệu trong database
"""
    sequelize db:seed:undo:all // Xóa dữ liệu mẫu nếu muốn tạo lại
    sequelize db:seed:undo --seed name-of-seed // Xóa dữ liệu mẫu theo tên
"""
